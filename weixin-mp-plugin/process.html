        "common_field_id_list": [
            "USER_FORM_INFO_FLAG_LOCATION",

3.然后调用 投放卡券创建二维码接口 取得领取会员卡的二维码

    "action_name": "QR_MULTIPLE_CARD",

                {
                    "card_id": "pKot_tzL0nH_DzZeHqfDMj7tF9RA",

            ]
        }
    }
}
